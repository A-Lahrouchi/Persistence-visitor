package edu.uga.miage.m1.polygons.gui.shapes;

import java.io.IOException;

import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;

public class Export {
    
    ObjectMapper objectMapper;
    String jsonShapesAsString;

    public Export() {
        this.objectMapper = new ObjectMapper();
        this.jsonShapesAsString = null;
    }

    public String getJsonShapesAsString(){
        return jsonShapesAsString;
    }

    public void writeJsonShapesAsString(JsonShapes jsonShapes) throws JsonProcessingException{
        jsonShapesAsString = objectMapper.writeValueAsString(jsonShapes);
    }
}
